# 
fig = plt.figure(figsize=(10, 4))
axes = fig.add_axes([0, 0, 1, 1])
axes.plot(china_grouped['confirmed'], label='   ', lw=3)
axes.plot(china_grouped['deaths'], label='  ', lw=3)
axes.plot(china_grouped['recovered'], label='   ', lw=3)
axes.plot(china_grouped['active'], label='   ', lw=3, linestyle='dashed')
